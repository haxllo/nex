2. **Replace the error message in `parse_text`** (line 1304-1305) with format-aware messages:
```rust
// Current (bad):
Err(ConfigError::Parse(format!(
    "invalid config format. toml error: {toml_err}; json error: {json_err}; json5 error: {json5_err}"

// Replace with:
fn format_parse_error(raw: &str, toml_err: &toml::de::Error, json_err: &serde_json::Error, json5_err: &json5::Error) -> String {
    // Detect format from raw text
    let looks_like_toml = raw.contains('[') || raw.contains(" = ");
    let looks_like_json = raw.trim_start().starts_with('{');
    
    if looks_like_toml {
        // TOML error: show line/column info
        if let Some(span) = toml_err.span() {
            format!("Config error at line {} column {}: {}. Expected format is TOML (key = \"value\").", 
                span.start, span.end, toml_err.message())
        } else {
            format!("Config error: {}. Check your config.toml syntax.", toml_err.message())
        }
    } else if looks_like_json {
        format!("Config error: {}. JSON config is still read but TOML is preferred. Use config.toml instead.", json_err)
    } else {
        format!("Config error: {}. Check your config file format — TOML is expected (key = \"value\").", toml_err.message())
    }
}
```

Replace the hardcoded `iced::window::Position::Centered` with cursor-aware positioning that places the overlay on whichever monitor the cursor is currently on.

**Specific changes:**

1. Add a function to get the monitor work area for the current cursor position:
```rust
#[cfg(target_os = "windows")]
fn cursor_monitor_work_area() -> Option<(i32, i32, i32, i32)> {
    use windows_sys::Win32::UI::WindowsAndMessaging::{GetCursorPos, MonitorFromPoint, GetMonitorInfoW, MONITORINFO, MONITOR_DEFAULTTONEAREST};
    use windows_sys::Win32::Foundation::{POINT, RECT};
    
    let mut cursor = POINT { x: 0, y: 0 };
    if unsafe { GetCursorPos(&mut cursor) } == 0 {
        return None;
    }
    let monitor = unsafe { MonitorFromPoint(cursor, MONITOR_DEFAULTTONEAREST) };
    if monitor.is_null() {
        return None;
    }
    let mut info = MONITORINFO {
        cbSize: std::mem::size_of::<MONITORINFO>() as u32,
        rcMonitor: RECT { left: 0, top: 0, right: 0, bottom: 0 },
        rcWork: RECT { left: 0, top: 0, right: 0, bottom: 0 },
        dwFlags: 0,
    };
    if unsafe { GetMonitorInfoW(monitor, &mut info) } == 0 {
        return None;
    }
    Some((info.rcWork.left, info.rcWork.top, info.rcWork.right, info.rcWork.bottom))
}

#[cfg(not(target_os = "windows"))]
fn cursor_monitor_work_area() -> Option<(i32, i32, i32, i32)> {
    None // Fall back to Centered on non-Windows
}
```

2. Compute the overlay position so it's centered on the cursor's monitor (vertically near top, like Spotlight/Alfred):
```rust
fn overlay_position() -> iced::window::Position {
    if let Some((left, top, right, bottom)) = cursor_monitor_work_area() {
        let monitor_width = (right - left) as f32;
        let monitor_height = (bottom - top) as f32;
        let x = left as f32 + (monitor_width - WINDOW_WIDTH) / 2.0;
        // Position at ~25% from top of work area (like macOS Spotlight)
        let y = top as f32 + monitor_height * 0.25;
        iced::window::Position::Specific(iced::Point::new(x, y.max(0.0)))
    } else {
        iced::window::Position::Centered
    }
}
```

3. Replace `position: iced::window::Position::Centered` in the `window_settings` (line 192) with `position: overlay_position()`.

4. **Do NOT** change the panel size or DPI scaling — wgpu/Iced handles DPI-aware rendering. The `WINDOW_WIDTH` and `panel_total_height()` use logical pixels (device-independent pixels) which Iced/winit scale automatically based on the monitor's DPI.
