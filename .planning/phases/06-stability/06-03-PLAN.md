**Better approach — use PowerShell for reliability:**
```rust
pub(crate) fn detect_hotkey_conflict_process(hotkey: &str) -> Option<String> {
    // Parse the hotkey to get modifier names and key for the message
    let parts: Vec<&str> = hotkey.split('+').map(|s| s.trim()).collect();
    if parts.is_empty() {
        return None;
    }
    
    // On Windows, there's no direct API to query which process registered a hotkey.
    // The best we can do is provide a helpful message listing common culprits
    // and suggest checking for apps that hook system-wide hotkeys.
    let key = parts.last()?;
    let common_conflicts: &[(&str, &[&str])] = &[
        ("PowerToys", &["Space", "F1", "F2"]), // PowerToys Run uses Alt+Space
        ("AHK scripts (AutoHotkey)", &["Space", "F"]),
        ("Discord", &["`"]),
        ("NVIDIA GeForce Experience", &["Z", "F1", "F2", "F3"]),
        ("AMD Adrenalin", &["R", "Z"]),
        ("Xbox Game Bar", &["G"]),
        ("Snipping Tool", &["S"]),
        ("OneNote", &["S", "N"]),
    ];
    
    let conflicts: Vec<&str> = common_conflicts.iter()
        .filter(|(_, keys)| keys.contains(&key))
        .map(|(app, _)| *app)
        .collect();
    
    if conflicts.is_empty() {
        Some(format!("Hotkey '{hotkey}' may be in use by another application. Check for applications that register global hotkeys (gaming overlays, screenshot tools, launchers)."))
    } else {
        Some(format!(
            "Hotkey '{hotkey}' is likely in use by: {}. Close these applications or change your Nex hotkey.",
            conflicts.join(", ")
        ))
    }
}
```

```rust
fn write_atomic(path: &Path, encoded: &str) -> Result<(), ConfigError> {
    // Write to a temp file in the same directory (required for atomic rename)
    let dir = path.parent().unwrap_or_else(|| Path::new("."));
    let temp_path = dir.join(format!(
        ".config.{}.tmp",
        SystemTime::now()
            .duration_since(UNIX_EPOCH)
            .map(|d| d.as_millis())
            .unwrap_or(0)
    ));
    
    // Write to temp file
    std::fs::write(&temp_path, encoded)?;
    
    // Atomic rename (on same filesystem, this is atomic on Windows NTFS)
    std::fs::rename(&temp_path, path).map_err(|e| {
        // Clean up temp file on failure
        let _ = std::fs::remove_file(&temp_path);
        ConfigError::Io(e)
    })
}
```
